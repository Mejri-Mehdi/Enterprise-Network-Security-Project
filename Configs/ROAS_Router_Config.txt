Cisco Router ROAS Configuration




enable
conf t

interface g0/0/0
  ip address 10.0.0.1 255.255.255.252
 no shutdown


! Remove old G0/0 config
interface g0/0/1
 no ip address
 no shutdown
exit

! Create subinterfaces
interface g0/0/1.10
 encapsulation dot1q 10
 ip address 192.168.10.254 255.255.255.0
 no shutdown
exit

interface g0/0/1.20
 encapsulation dot1q 20
 ip address 192.168.20.254 255.255.255.0
 no shutdown
exit

interface g0/0/1.30
 encapsulation dot1q 30
 ip address 192.168.30.254 255.255.255.0
 no shutdown
exit

interface g0/0/1.40
 encapsulation dot1q 40
 ip address 192.168.40.254 255.255.255.0
 no shutdown
exit

interface g0/0/1.50
 encapsulation dot1q 50
 ip address 192.168.50.254 255.255.255.0
 no shut
exit

interface g0/0/1.99
 encapsulation dot1q 99 native
 ip address 192.168.99.1 255.255.255.0
 no shutdown
exit

! Default route to Firewall
ip route 0.0.0.0 0.0.0.0 10.0.0.2

end
write
! 5. (Optional Professional Step) 
! Secure your router against unauthorized access
line console 0
 password C1sco!Secure#2026
 login
 exit
line vty 0 4
 password C1sco!Secure#2026
 login
 transport input ssh
 exit
enable secret C1sco!Secure#2026
service password-encryption
exit
